		$('rating').onmouseout = function() { show_current_rating(); }
	});
	
	function show_current_rating() {
		var whole = Math.floor(current_rating);
		var half = current_rating - whole;
					
		$$('#rating .star').each(function(x) {
			x.removeClassName('full');
			x.removeClassName('half');
		});
		for (i = 0; i < whole; i++) {
			$$('#rating .star')[i].addClassName('full');
		}
		if (half > 0) $$('#rating .star')[whole].addClassName('half');
	}
